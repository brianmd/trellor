class Trellor {
  self.singleton()
  -- queries --
  board_names()
  list_names(board_regex)
  cart_names(board_regex,list_regex)
  create_card(board,list,card,descript)
  -- private queries --
  boards()
  boards=(boards)
  find_board(board_regex)
  find_list(board_regex,list_regex)
  cards(board_regex,list_regex)
  find_card(board,list,card)
  -- connecting --
  client()
  connect()
  user()
}
